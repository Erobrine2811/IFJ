// Correct: Hexadecimal literals
import "ifj25" for Ifj
class Program {
    static main() {
        var hex1 
        hex1 = 0xFF
        Ifj.write("0xFF = ")
        Ifj.write(hex1)
        Ifj.write("\n")
        
        var hex2 
        hex2 = 0xABCD
        Ifj.write("0xABCD = ")
        Ifj.write(hex2)
        Ifj.write("\n")
        
        var result 
        result = 0x10 + 0x20
        Ifj.write("0x10 + 0x20 = ")
        Ifj.write(result)
        Ifj.write("\n")
    }
}
