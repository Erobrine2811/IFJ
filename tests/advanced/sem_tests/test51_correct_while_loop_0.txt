// Correct: While loop with counter
import "ifj25" for Ifj
class Program {
    static main() {
        var counter 
        counter = 0
        var sum 
        sum = 0
        
        while (counter < 5) {
            sum = sum + counter
            Ifj.write("Counter: ")
            Ifj.write(counter)
            Ifj.write(", Sum: ")
            Ifj.write(sum)
            Ifj.write("\n")
            counter = counter + 1
        }
        
        Ifj.write("Final sum: ")
        Ifj.write(sum)
        Ifj.write("\n")
    }
}
