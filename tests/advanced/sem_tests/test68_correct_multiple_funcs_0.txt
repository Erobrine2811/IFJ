// Correct: Multiple function definitions
import "ifj25" for Ifj
class Program {
    static add(a, b) {
        return a + b
    }
    
    static multiply(a, b) {
        return a * b
    }
    
    static square(x) {
        return x*x
    }
    
    static main() {
        var x 
        x = 5
        var sum 
        sum = add(x, 3)
        var sq 
        sq = square(x)
        
        Ifj.write("Sum: ")
        Ifj.write(sum)
        Ifj.write("\n")
        Ifj.write("Square: ")
        Ifj.write(sq)
        Ifj.write("\n")
    }
}
