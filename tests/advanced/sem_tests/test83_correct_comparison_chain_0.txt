import "ifj25" for Ifj

    static main() {
        var a 
        a = 5
        var b 
        b = 10
        
        if (a < b) {
            Ifj.write("a is less than b\n")
        } else {
            Ifj.write("a is not less\n")
        }
        
        if (a <= b) {
            Ifj.write("a is less or equal\n")
        } else {
            Ifj.write("a is greater\n")
        }
        
        if (a == a) {
            Ifj.write("a equals itself\n")
        } else {
            Ifj.write("impossible\n")
        }
    }
}
