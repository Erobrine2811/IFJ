// Correct: Float and exponential notation
import "ifj25" for Ifj
class Program {
    static main() {
        var pi 
        pi = 3.14159
        Ifj.write("Pi: ")
        Ifj.write(pi)
        Ifj.write("\n")
        
        var e 
        e = 2.71828
        Ifj.write("e: ")
        Ifj.write(e)
        Ifj.write("\n")
        
        var sci 
        sci = 1.5e3
        Ifj.write("1.5e3 = ")
        Ifj.write(sci)
        Ifj.write("\n")
        
        var small 
        small = 2.5e-2
        Ifj.write("2.5e-2 = ")
        Ifj.write(small)
        Ifj.write("\n")
    }
}
