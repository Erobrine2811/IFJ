// Correct: Complex while loop with nested blocks
import "ifj25" for Ifj
class Program {
    static main() {
        var i 
        i = 0
        var sum 
        sum = 0
        
        while (i < 10) {
            var temp 
            temp = i * 2
            
            if (temp > 10) {
                sum = sum + temp
            } else {
                sum = sum + i
            }
            
            Ifj.write("i: ")
            Ifj.write(i)
            Ifj.write(", sum: ")
            Ifj.write(sum)
            Ifj.write("\n")
            
            i = i + 1
        }
        
        Ifj.write("Final sum: ")
        Ifj.write(sum)
        Ifj.write("\n")
    }
}
