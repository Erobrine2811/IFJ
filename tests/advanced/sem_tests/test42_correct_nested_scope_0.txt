// Correct: Variable shadowing in nested scope
import "ifj25" for Ifj
class Program {
    static main() {
        var x 
        x = 5
        Ifj.write("Outer x: ")
        Ifj.write(x)
        Ifj.write("\n")
        {
            var x 
            x = 10
            Ifj.write("Inner x: ")
            Ifj.write(x)
            Ifj.write("\n")
        }
        Ifj.write("Outer x again: ")
        Ifj.write(x)
        Ifj.write("\n")
    }
}
